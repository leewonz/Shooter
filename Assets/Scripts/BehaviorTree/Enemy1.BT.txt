tree("Idle")
	repeat
		sequence
			mute 
				fallback
					SetRandomPath(2.0,6.0,10)
					SetRandomPath(1.0,2.0,10)
			mute
				while not IsInDestination
					parallel
						repeat MoveToPosition
						repeat RotateToDestination(360.0)
						not Wait 2.0
			random
				Wait 1.0
				Succeed

tree("Attack")
	mute 
		sequence
			mute
				parallel
					repeat MoveToPosition
					not WaitRandom(0.1,0.2)
			parallel
				repeat
					sequence
						repeat 2
							sequence
								not repeat not SetRandomAttackPath(1.5,6.0,2.0,10)
						
								mute
									while not IsInDestination
										parallel
											repeat MoveToPosition
											not Wait 2.0
						sequence
							Wait 0.4
							FireAll()
				repeat RotateToPlayer(180.0)
